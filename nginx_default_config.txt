# 2. Beta App (Customer, Vendor, Admin, API) - HTTPS
server {
    listen 443 ssl;
    server_name beta.resortwala.com;

    ssl_certificate /etc/letsencrypt/live/resortwala.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/resortwala.com/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;
    
    # Root points to Customer App
    root /var/www/html/staging.resortwala.com;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }

    # Vendor App
    location /vendor {
        alias /var/www/html/stagingvendor.resortwala.com;
        try_files $uri $uri/ /vendor/index.html;
    }

    # Admin App
    location /admin {
        alias /var/www/html/stagingadmin.resortwala.com;
        try_files $uri $uri/ /admin/index.html;
    }

    # API
    location ^~ /api {
        alias /var/www/html/stagingapi.resortwala.com/public;
        try_files $uri $uri/ @laravel;
    }

    # Serve Storage Files from API
    location ^~ /storage {
        alias /var/www/html/stagingapi.resortwala.com/public/storage;
        try_files $uri $uri/ =404;
    }

    location @laravel {
        rewrite /api/(.*)$ /api/$1 break;
        fastcgi_pass unix:/var/run/php/php8.2-fpm.sock;
        fastcgi_index index.php;
        include fastcgi_params;
        fastcgi_param SCRIPT_FILENAME /var/www/html/stagingapi.resortwala.com/public/index.php;
    }
}
